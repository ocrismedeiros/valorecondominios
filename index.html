⚙️ Setup — Criar Tabelas no Supabase

Execute o SQL abaixo no SQL Editor do seu projeto Supabase.

1
Acesse supabase.com → seu projeto → SQL Editor → New query
2
Cole e execute o SQL clicando em RUN
CREATE TABLE IF NOT EXISTS task_completions (id uuid DEFAULT gen_random_uuid() PRIMARY KEY, collaborator text NOT NULL, task_key text NOT NULL, month_year text NOT NULL, done boolean DEFAULT false, completed_at timestamptz, updated_at timestamptz DEFAULT now(), UNIQUE(collaborator, task_key, month_year)); ALTER TABLE task_completions ENABLE ROW LEVEL SECURITY; CREATE POLICY "public_access" ON task_completions FOR ALL TO anon USING (true) WITH CHECK (true); ALTER publication supabase_realtime ADD TABLE task_completions; CREATE TABLE IF NOT EXISTS boleto_steps (id uuid DEFAULT gen_random_uuid() PRIMARY KEY, condo_key text NOT NULL, step_id text NOT NULL, month_year text NOT NULL, done boolean DEFAULT false, completed_at timestamptz, updated_at timestamptz DEFAULT now(), UNIQUE(condo_key, step_id, month_year)); ALTER TABLE boleto_steps ENABLE ROW LEVEL SECURITY; CREATE POLICY "public_access" ON boleto_steps FOR ALL TO anon USING (true) WITH CHECK (true); ALTER publication supabase_realtime ADD TABLE boleto_steps; CREATE TABLE IF NOT EXISTS leitura_steps (id uuid DEFAULT gen_random_uuid() PRIMARY KEY, condo_key text NOT NULL, step_id text NOT NULL, month_year text NOT NULL, done boolean DEFAULT false, completed_at timestamptz, updated_at timestamptz DEFAULT now(), UNIQUE(condo_key, step_id, month_year)); ALTER TABLE leitura_steps ENABLE ROW LEVEL SECURITY; CREATE POLICY "public_access" ON leitura_steps FOR ALL TO anon USING (true) WITH CHECK (true); ALTER publication supabase_realtime ADD TABLE leitura_steps;
3
Aguarde "Success" e clique abaixo:
